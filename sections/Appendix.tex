    &\int x\cos (ax)dx=\frac{1}{a^2}[\cos(ax)+ax\sin(ax)]\\
    &\int x^2\sin(ax)=\frac{1}{a^3}[-(a^2x-2)\cos(ax)+2ax\sin(ax)]\\

\end{aligned}$$
